2. **更新 App Store** `src/store/appStore.ts`

```typescript
import { create } from 'zustand';

interface Pet {
  id: string;
  name: string;
  avatar?: string;
  type: 'dog' | 'cat' | 'other';
  breed: string;
  birthday: string;
  weight: number;
}

interface AppStore {
  currentPet: Pet | null;
  pets: Pet[];
  bondScore: number;
  setCurrentPet: (pet: Pet) => void;
  addPet: (pet: Pet) => void;
  updateBondScore: (score: number) => void;
}

export const useAppStore = create<AppStore>((set) => ({
  currentPet: null,
  pets: [],
  bondScore: 85,
  setCurrentPet: (pet) => set({ currentPet: pet }),
  addPet: (pet) => set((state) => ({ pets: [...state.pets, pet] })),
  updateBondScore: (score) => set({ bondScore: score }),
}));
```

```tsx
// src/components/health/FAB.tsx
import React, { useState } from 'react';
import { motion, AnimatePresence } from 'framer-motion';
import { Plus, Type, Mic, Camera, Video, FileText } from 'lucide-react';

export function FAB({ onAddRecord }: { onAddRecord: (type: string) => void }) {
  const [isOpen, setIsOpen] = useState(false);

  const actions = [
    { icon: Type, label: '文字', type: 'text', color: 'primary' },
    { icon: Mic, label: '语音', type: 'voice', color: 'purple' },
    { icon: Camera, label: '照片', type: 'photo', color: 'success' },
    { icon: Video, label: '视频', type: 'video', color: 'warning' },
    { icon: FileText, label: '文件', type: 'file', color: 'secondary' },
  ];

  return (
    <div className="fixed bottom-8 right-8 z-50">
      <AnimatePresence>
        {isOpen &amp;&amp; actions.map((action, index) => (
          <motion.button
            key={action.type}
            initial={{ opacity: 0, y: 20, scale: 0 }}
            animate={{ opacity: 1, y: -60 - index * 56, scale: 1 }}
            exit={{ opacity: 0, y: 0, scale: 0 }}
            className={`absolute w-12 h-12 rounded-full bg-${action.color}-500 text-white flex items-center justify-center shadow-lg`}
            onClick={() => {
              onAddRecord(action.type);
              setIsOpen(false);
            }}
          >
            <action.icon size={20} />
          </motion.button>
        ))}
      </AnimatePresence>
      
      <motion.button
        animate={{ rotate: isOpen ? 45 : 0 }}
        className="w-14 h-14 rounded-full bg-primary-500 text-white flex items-center justify-center shadow-xl"
        onClick={() => setIsOpen(!isOpen)}
      >
        <Plus size={28} />
      </motion.button>
    </div>
  );
}
```
