```typescript
// backend/src/routes/community.ts
import { Router, Request, Response } from 'express';
import { body, validationResult } from 'express-validator';
import prisma from '../lib/prisma';
import { authenticateToken } from '../middleware';

const router = Router();
router.use(authenticateToken);

// 发布动态
router.post('/posts', [
  body('content').isLength({ min: 1 }),
  body('petId').optional().isUUID()
], async (req: Request, res: Response) =&gt; {
  const errors = validationResult(req);
  if (!errors.isEmpty()) {
    return res.status(400).json({ errors: errors.array() });
  }
  
  const { content, petId, images, videos, tags } = req.body;
  
  const post = await prisma.communityPost.create({
    data: {
      userId: req.userId!,
      petId,
      content,
      images: images || [],
      videos: videos || [],
      tags: tags || []
    },
    include: { user: true, pet: true }
  });
  
  res.status(201).json({ post });
});

// 获取动态列表
router.get('/posts', async (req: Request, res: Response) =&gt; {
  const { page = 1, limit = 20, type = 'all', petId } = req.query;
  
  const where: any = {};
  if (petId) where.petId = petId;
  if (type !== 'all') where.type = type;
  
  const posts = await prisma.communityPost.findMany({
    where,
    include: { user: true, pet: true },
    orderBy: { createdAt: 'desc' },
    skip: (page - 1) * limit,
    take: Number(limit)
  });
  
  res.json({ posts });
});

// 点赞
router.post('/posts/:id/like', async (req: Request, res: Response) =&gt; {
  const like = await prisma.postLike.create({
    data: {
      postId: req.params.id,
      userId: req.userId!
    }
  });
  res.json({ like });
});

// 评论
router.post('/posts/:id/comments', [
  body('content').isLength({ min: 1 })
], async (req: Request, res: Response) =&gt; {
  const comment = await prisma.postComment.create({
    data: {
      postId: req.params.id,
      userId: req.userId!,
      content: req.body.content
    },
    include: { user: true }
  });
  res.status(201).json({ comment });
});

// 宠物匹配
router.get('/matches', async (req: Request, res: Response) =&gt; {
  // 简单的匹配算法示例
  const pets = await prisma.pet.findMany({
    where: { 
      NOT: { userId: req.userId },
      gender: { not: null } // 简单性别匹配
    },
    include: { user: true },
    take: 20
  });
  res.json({ matches: pets });
});

// 附近宠友
router.get('/nearby', async (req: Request, res: Response) =&gt; {
  const { lat, lng, radius = 5 } = req.query;
  // 简化的附近查询
  const users = await prisma.user.findMany({
    where: { NOT: { id: req.userId } },
    include: { pets: true },
    take: 30
  });
  res.json({ users });
});

export default router;
```
